11  import io.netty.util.AttributeKey;
12  
13  /**
14   * Inbound handler.
15   * 
16   * @author Tom Parker
17   *
18   */
19  public class WWSClientHandler extends ChannelInboundHandlerAdapter {
20    @SuppressWarnings("unused")
21    private static final Logger LOGGER = LoggerFactory.getLogger(WWSClient.class);
22  
23    /** handler key */
24    public static final AttributeKey<AbstractCommandHandler> handlerKey =
25        AttributeKey.valueOf("commandHandler");
26  
27    @Override
28    public void channelRead(ChannelHandlerContext ctx, Object msg) throws IOException {
29      AbstractCommandHandler handler = ctx.channel().attr(handlerKey).get();
30      handler.handle(msg);
31    }
32  
33    @Override
34    public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) {
35      cause.printStackTrace();
36      ctx.close();
37    }
38  }
